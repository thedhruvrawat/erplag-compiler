** Takes input value, prints the input value, followed by two times the input value **
<<module compute_expr>>
takes input[ a: integer];
returns [u:integer];
start
    print(a);
    u := a * 2;
	print(u);
end

<<<driver program>>>
start
	declare x:integer;
	get_value(x);
	print(x);
	[x]:= use module compute_expr with parameters x;				
	print(x);
end
